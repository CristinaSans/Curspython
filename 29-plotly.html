import pandas as pd
import plotly.express as px
df= pd.read_csv('titanic.csv',sep=',', decimal='.', index_col=0)
df['AgeGroup'] = df['Age'].apply(lambda x: 'Menor' if x < 18 else 'Adult')
survival_pct = (
    df.groupby(['Pclass', 'AgeGroup'])['Survived']
      .mean()
      .reset_index()
)

survival_pct['Survived'] *= 100
fig = px.bar( survival_pct, x='Pclass', y='Survived', color='AgeGroup', barmode='group', # tamb pots provar 'stack' 
              color_discrete_sequence= px.colors.qualitative.Dark24,
              labels={'Survived': 'Percentatge de supervivncia (%)'}, 
              title='Percentatge de supervivncia per classe i grup dedat' ) 
fig.show()

fig = px.imshow( survival_pct.pivot(index='Pclass', columns='AgeGroup', values='Survived'), 
                 labels=dict(x="Grup d'edat", y="Classe", color="Supervivncia (%)"),
                 text_auto=".1f", color_continuous_scale=["#ffa600","#ef5675", "#7a5195", "#003f5c"], 
                 title="Percentatge de supervivncia per classe i grup dedat" )
fig.show()

import pandas as pd
import plotly.graph_objects as go

# Carregar dades
df = pd.read_csv('cotizacion.csv', sep=';', decimal=',', thousands='.', index_col=0)

# Crear columna d'obertura (Open) a partir del tancament del dia anterior
df['Open'] = df['Final'].shift(1)

# Eliminar primer registre (Open = NaN)
df = df.dropna(subset=['Open'])

# Crear figura amb subgrfics
fig = go.Figure()

# --- Candlestick ---
fig.add_trace(go.Candlestick(
    x=df.index,
    open=df['Open'],
    high=df['Mximo'],
    low=df['Mnimo'],
    close=df['Final'],
    increasing_line_color='green',
    decreasing_line_color='red',
    name='Preu'
))

# --- Volum ---
fig.add_trace(go.Bar(
    x=df.index,
    y=df['Volumen'],
    name='Volum',
    marker_color='plum',
    opacity=0.5,
    yaxis='y2'
))

# Configuraci dels eixos
fig.update_layout(
    title='Candlestick + Volum',
    xaxis=dict(title='Data'),
    yaxis=dict(title='Preu'),
    yaxis2=dict(
        title='Volum',
        overlaying='y',
        side='right',
        showgrid=False
    ),
    xaxis_rangeslider_visible=False
)

fig.show()
