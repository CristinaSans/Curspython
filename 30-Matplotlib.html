import matplotlib.pyplot as plt
import numpy as np

# Datos
edades = ["De 16 a 19","De 20 a 24"," De 25 a 29","De 30 a 34","De 35 a 39"," De 40 a 44"]

datos = {
    "Ambos sexos": ["39,90","35,20","51,70","62,50","60,40","66,40"],
    "Hombres":     ["48,70","45,30","66,30","76,10","75,50","75,40"],
    "Mujeres":     ["23,70","24,40","33,80","45,80","45,10","54,80"]
}

# Conversin coma  punto
ambos = [float(x.replace(",", ".")) for x in datos["Ambos sexos"]]
hombres = [float(x.replace(",", ".")) for x in datos["Hombres"]]
mujeres = [float(x.replace(",", ".")) for x in datos["Mujeres"]]

# Posiciones
x = np.arange(len(edades))
width = 0.25  # ancho de cada barra

plt.figure(figsize=(10,6))

plt.bar(x - width, mujeres, width, color='red', label='Mujeres')
plt.bar(x,         hombres, width, color='blue', label='Hombres')
plt.bar(x + width, ambos,   width, color='green', label='Ambos sexos')

plt.xticks(x, edades, rotation=30)
plt.ylabel("Porcentaje")
plt.title("% de estudios primarios incompletos por sexo y grupo de edad")
plt.legend(loc='upper left', bbox_to_anchor=(1.1, 0.9))
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import numpy as np
import random

# Datos originales
meses = ['Gen', 'Feb', 'Mar', 'Abr', 'Mayo', 'Junio']
temperaturas = [12.5, 12.3, 15.2, 16.4, 20.5, 22.2]
plujamm = [335, 203, 334, 604, 700, 555]

# Nuevos meses
nuevos_meses = ['Jul', 'Ago', 'Sep', 'Oct', 'Nov', 'Dic']

# Generar valores aleatorios coherentes
nuevas_temperaturas = [round(random.uniform(10, 30), 1) for _ in nuevos_meses]
nueva_pluja = [random.randint(150, 800) for _ in nuevos_meses]

# Unir todo
meses.extend(nuevos_meses)
temperaturas.extend(nuevas_temperaturas)
plujamm.extend(nueva_pluja)

print("Meses:", meses)
print("Temperaturas:", temperaturas)
print("Pluja (mm):", plujamm)

# Tamao proporcional a la lluvia
tamanos = [p * 0.5 for p in plujamm]

# Paleta estacional
colores = [
    '#4A90E2', '#6FB1FC',   # Invierno (Ene, Feb)
    '#7ED321', '#4CAF50', '#2E7D32',  # Primavera (Mar, Abr, May)
    '#F5D76E', '#F4A742', '#F39C12',  # Verano (Jun, Jul, Ago)
    '#D35400', '#A04000', '#6E2C00',  # Otoo (Sep, Oct, Nov)
    '#357ABD'  # Invierno (Dic)
]

plt.figure(figsize=(10,6))

# Grfico principal
plt.scatter(meses, temperaturas, s=tamanos, c=colores)

# Crear leyenda por estacin usando "handles" falsos
invierno = plt.Line2D([0], [0], marker='o', color='w', label='Invierno',
                      markerfacecolor='#4A90E2', markersize=10)
primavera = plt.Line2D([0], [0], marker='o', color='w', label='Primavera',
                       markerfacecolor='#7ED321', markersize=10)
verano = plt.Line2D([0], [0], marker='o', color='w', label='Verano',
                    markerfacecolor='#F5D76E', markersize=10)
otoo = plt.Line2D([0], [0], marker='o', color='w', label='Otoo',
                   markerfacecolor='#D35400', markersize=10)

plt.legend(handles=[invierno, primavera, verano, otoo],
           title="Estaciones", loc='upper left', bbox_to_anchor=(1.05, 1))

plt.title("Temperatura media y volumen de lluvia por mes (paleta estacional)")

import matplotlib.pyplot as plt
import numpy as np

from matplotlib import cm

plt.style.use('_mpl-gallery')

n_radii = 8
n_angles = 36

# Make radii and angles spaces
radii = np.linspace(0.125, 1.0, n_radii)
angles = np.linspace(0, 2*np.pi, n_angles, endpoint=False)[..., np.newaxis]

# Convert polar (radii, angles) coords to cartesian (x, y) coords.
x = np.append(0, (radii*np.cos(angles)).flatten())
y = np.append(0, (radii*np.sin(angles)).flatten())
z = np.sin(-x*y)

# Plot
fig, ax = plt.subplots(figsize= (10, 8),subplot_kw={'projection': '3d'})
ax.plot_trisurf(x, y, z, vmin=z.min() * 2, cmap=cm.Reds)

ax.set(xticklabels=[],
       yticklabels=[],
       zticklabels=[])

plt.show()
